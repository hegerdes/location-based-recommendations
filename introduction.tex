	\begin{frame}{Definition}
		\begin{quote}
			Location-based recommendation is a recommender system that incorporates \textcolor{red}{location information} into algorithms to attempt to provide more \textcolor{red}{relevant recommendations} to users.
		\end{quote}
		\begin{flushright}
			---~Wikipedia
		\end{flushright}

		\uncover<2->{
			\begin{center}
				\large\textbf{Standortdaten \bm{$\Rightarrow$} Relevante Empfehlungen}
			\end{center}
		}
	\end{frame}

	\begin{frame}{In diesem Projekt}
		Welche Daten könnte man für LBR nutzen?
		\begin{itemize}
			\item GPS-Position
			\item Event-Filter
			\item User-Teilnahme an ähnlichen Events in der Vergangenheit
			\item Popularität bestimmter Locations in der Umgebung (mithilfe von historischen oder Echtzeit-Daten)
		\end{itemize}
	\end{frame}

	\begin{frame}{User Stories}
		\userstory{Ich möchte Zeit sparen, indem ich Veranstaltungen, die mich interessieren, schnell finden kann.}

		\userstory{Ich möchte festlegen können, für welche Art von Events ich benachrichtigt werde.}
	\end{frame}

	\begin{frame}{Projektziel}
		\textbf{Must-Haves:}
		\begin{itemize}
			\item Geordnete Liste passender Veranstaltungen
			\item Push-Nachrichten, wenn geeignete Events in der Nähe stattfinden
			\item Konfiguration von Präferenzen, Suchradius, Push-Nachrichten etc.
		\end{itemize}
		
		\textbf{Nice to have:}
		\begin{itemize}
			\item Hervorhebung empfohlener Events auf der Karte
		\end{itemize}
	\end{frame}
